261  {
262  // Get the i-th detected tag
263  apriltag_detection_t *detection;
264  zarray_get(detections_, i, &detection);
265 
266  // Bootstrap this for loop to find this tag's description amongst
267  // the tag bundles. If found, add its points to the bundle's set of
268  // object-image corresponding points (tag corners) for cv::solvePnP.
269  // Don't yet run cv::solvePnP on the bundles, though, since we're still in
270  // the process of collecting all the object-image corresponding points
271  int tagID = detection->id;
272  bool is_part_of_bundle = false;
273  for (unsigned int j = 0; j < tag_bundle_descriptions_.size(); j++)
274  {
275  // Iterate over the registered bundles
276  TagBundleDescription bundle = tag_bundle_descriptions_[j];
277 
278  if (bundle.id2idx_.find(tagID) != bundle.id2idx_.end())

281  // the bundle description)

283  std::string bundleName = bundle.name();
284

331  std::vector<cv::Point2d> standaloneTagImagePoints;
332  addObjectPoints(tag_size / 2, cv::Matx44d::eye(), standaloneTagObjectPoints);
333  addImagePoints(detection, standaloneTagImagePoints);
334  Eigen::Matrix4d transform = getRelativeTransform(standaloneTagObjectPoints,
335  standaloneTagImagePoints,

341  makeTagPose(transform, rot_quaternion, image->header);

347  tag_detection.size.push_back(tag_size);
348  tag_detection_array.detections.push_back(tag_detection);

351  }

354  // Estimate bundle origin pose for each bundle in which at least one
355  // member tag was detected
356

358  {
359  // Get bundle name
360  std::string bundleName = tag_bundle_descriptions_[j].name();
361 
362  std::map<std::string,
363  std::vector<cv::Point3d>>::iterator it =
364  bundleObjectPoints.find(bundleName);
365  if (it != bundleObjectPoints.end())

370 
371  Eigen::Matrix4d transform =
372  getRelativeTransform(bundleObjectPoints[bundleName],

374  Eigen::Matrix3d rot = transform.block(0, 0, 3, 3);
375  Eigen::Quaternion<double> rot_quaternion(rot);

378  makeTagPose(transform, rot_quaternion, image->header);
379 
380  // Add the detection to the back of the tag detection array
381  AprilTagDetection tag_detection;
382  tag_detection.pose = bundle_pose;
383  tag_detection.id = bundle.bundleIds();
384  tag_detection.size = bundle.bundleSizes();
385  tag_detection_array.detections.push_back(tag_detection);
386  detection_names.push_back(bundle.name());

388  // for (int k = 0; k < tag_detection.id.size(); k++)
389  // std::cout << tag_detection.id[k]<<" ";
390  // std::cout << std::endl;
391  }
392  }
393

395  if (publish_tf_)
396  {
397  for (unsigned int i = 0; i < tag_detection_array.detections.size(); i++)
398  {
399  geometry_msgs::PoseStamped pose;
400  pose.pose = tag_detection_array.detections[i].pose.pose.pose;
401  pose.header = tag_detection_array.detections[i].pose.header;
402  tf::Stamped<tf::Transform> tag_transform;
403  tf::poseStampedMsgToTF(pose, tag_transform);
404  tf_pub_.sendTransform(tf::StampedTransform(tag_transform,
405  tag_transform.stamp_,
406  camera_tf_frame_,
407  detection_names[i]));
408  }
409  }
410 
411  return tag_detection_array;

655  void TagDetector::removeDuplicates()
656  {
657  zarray_sort(detections_, &idComparison);
658  int count = 0;
659  bool duplicate_detected = false;
660  while (true)
661  {
662  if (count > zarray_size(detections_) - 1)
663  {
664  // The entire detection set was parsed
665  return;
666  }
667  apriltag_detection_t *detection;
668  zarray_get(detections_, count, &detection);
669  int id_current = detection->id;
670  // Default id_next value of -1 ensures that if the last detection
671  // is a duplicated tag ID, it will get removed
672  int id_next = -1;
673  if (count < zarray_size(detections_) - 1)
674  {
675  zarray_get(detections_, count + 1, &detection);
676  id_next = detection->id;
677  }
678  if (id_current == id_next || (id_current != id_next && duplicate_detected))
679  {
680  duplicate_detected = true;
681  // Remove the current tag detection from detections array
682  int shuffle = 0;
683  zarray_remove_index(detections_, count, shuffle);
684  if (id_current != id_next)
685  {
686  ROS_WARN_STREAM("Pruning tag ID " << id_current << " because it "
687  "appears more than once in the image.");
688  duplicate_detected = false; // Reset
689  }
690  continue;
691  }
692  else
693  {
694  count++;
695  }
696  }
697  }

732  Eigen::Matrix4d TagDetector::getRelativeTransform(
733  std::vector<cv::Point3d> objectPoints,
734  std::vector<cv::Point2d> imagePoints,
735  double fx, double fy, double cx, double cy) const
736  {
737  // perform Perspective-n-Point camera pose estimation using the
738  // above 3D-2D point correspondences
739  cv::Mat rvec, tvec;
740  cv::Matx33d cameraMatrix(fx, 0, cx,
741  0, fy, cy,
742  0, 0, 1);
743  cv::Vec4f distCoeffs(0, 0, 0, 0); // distortion coefficients
744  // TODO Perhaps something like SOLVEPNP_EPNP would be faster? Would
745  // need to first check WHAT is a bottleneck in this code, and only
746  // do this if PnP solution is the bottleneck.
747  cv::solvePnP(objectPoints, imagePoints, cameraMatrix, distCoeffs, rvec, tvec);
748  cv::Matx33d R;
749  cv::Rodrigues(rvec, R);
750  Eigen::Matrix3d wRo;
751  wRo << R(0, 0), R(0, 1), R(0, 2), R(1, 0), R(1, 1), R(1, 2), R(2, 0), R(2, 1), R(2, 2);
752 
753  Eigen::Matrix4d T; // homogeneous transformation matrix
754  T.topLeftCorner(3, 3) = wRo;
755  T.col(3).head(3) << tvec.at<double>(0), tvec.at<double>(1), tvec.at<double>(2);
756  T.row(3) << 0, 0, 0, 1;
757  return T;
758  }

896  std::vector<TagBundleDescription> TagDetector::parseTagBundles(
897  XmlRpc::XmlRpcValue &tag_bundles)
898  {
899  std::vector<TagBundleDescription> descriptions;
900  ROS_ASSERT(tag_bundles.getType() == XmlRpc::XmlRpcValue::TypeArray);
901 
902  // Loop through all tag bundle descritions
903  for (int32_t i = 0; i < tag_bundles.size(); i++)
904  {
905  ROS_ASSERT(tag_bundles[i].getType() == XmlRpc::XmlRpcValue::TypeStruct);
906  // i-th tag bundle description
907  XmlRpc::XmlRpcValue &bundle_description = tag_bundles[i];
908 
909  std::string bundleName;
910  if (bundle_description.hasMember("name"))
911  {
912  ROS_ASSERT(bundle_description["name"].getType() ==
913  XmlRpc::XmlRpcValue::TypeString);
914  bundleName = (std::string)bundle_description["name"];
915  }
916  else
917  {
918  std::stringstream bundle_name_stream;
919  bundle_name_stream << "bundle_" << i;
920  bundleName = bundle_name_stream.str();
921  }
922  TagBundleDescription bundle_i(bundleName);
923  ROS_INFO("Loading tag bundle '%s'", bundle_i.name().c_str());
924 
925  ROS_ASSERT(bundle_description["layout"].getType() ==
926  XmlRpc::XmlRpcValue::TypeArray);
927  XmlRpc::XmlRpcValue &member_tags = bundle_description["layout"];
928 
929  // Loop through each member tag of the bundle
930  for (int32_t j = 0; j < member_tags.size(); j++)
931  {
932  ROS_ASSERT(member_tags[j].getType() == XmlRpc::XmlRpcValue::TypeStruct);
933  XmlRpc::XmlRpcValue &tag = member_tags[j];
934 
935  ROS_ASSERT(tag["id"].getType() == XmlRpc::XmlRpcValue::TypeInt);
936  int id = tag["id"];
937 
938  ROS_ASSERT(tag["size"].getType() == XmlRpc::XmlRpcValue::TypeDouble);
939  double size = tag["size"];
940 
941  // Make sure that if this tag was specified also as standalone,
942  // then the sizes match
943  StandaloneTagDescription *standaloneDescription;
944  if (findStandaloneTagDescription(id, standaloneDescription, false))
945  {
946  ROS_ASSERT(size == standaloneDescription->size());
947  }
948 
949  // Get this tag's pose with respect to the bundle origin
950  double x = xmlRpcGetDoubleWithDefault(tag, "x", 0.);
951  double y = xmlRpcGetDoubleWithDefault(tag, "y", 0.);
952  double z = xmlRpcGetDoubleWithDefault(tag, "z", 0.);
953  double qw = xmlRpcGetDoubleWithDefault(tag, "qw", 1.);
954  double qx = xmlRpcGetDoubleWithDefault(tag, "qx", 0.);
955  double qy = xmlRpcGetDoubleWithDefault(tag, "qy", 0.);
956  double qz = xmlRpcGetDoubleWithDefault(tag, "qz", 0.);
957  Eigen::Quaterniond q_tag(qw, qx, qy, qz);
958  q_tag.normalize();
959  Eigen::Matrix3d R_oi = q_tag.toRotationMatrix();
960 
961  // Build the rigid transform from tag_j to the bundle origin
962  cv::Matx44d T_mj(R_oi(0, 0), R_oi(0, 1), R_oi(0, 2), x,
963  R_oi(1, 0), R_oi(1, 1), R_oi(1, 2), y,
964  R_oi(2, 0), R_oi(2, 1), R_oi(2, 2), z,
965  0, 0, 0, 1);
966 
967  // Register the tag member
968  bundle_i.addMemberTag(id, size, T_mj);
969  ROS_INFO_STREAM(" " << j << ") id: " << id << ", size: " << size << ", "
970  << "p = [" << x << "," << y << "," << z << "], "
971  << "q = [" << qw << "," << qx << "," << qy << ","
972  << qz << "]");
973  }
974  descriptions.push_back(bundle_i);
975  }
976  return descriptions;
